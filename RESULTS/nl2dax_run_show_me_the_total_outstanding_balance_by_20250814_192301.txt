show me the total outstanding balance by customer type

========== GENERATED SQL ==========
SELECT 
    c.CUSTOMER_TYPE_DESCRIPTION AS customer_type,
    SUM(f.OUTSTANDING_BALANCE) AS total_outstanding_balance
FROM FIS_CA_DETAIL_FACT AS f
INNER JOIN FIS_CUSTOMER_DIMENSION AS c
    ON f.CUSTOMER_KEY = c.CUSTOMER_KEY
GROUP BY 
    c.CUSTOMER_TYPE_DESCRIPTION
ORDER BY 
    c.CUSTOMER_TYPE_DESCRIPTION;
