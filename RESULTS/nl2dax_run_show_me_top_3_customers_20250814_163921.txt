
========== NATURAL LANGUAGE QUERY ==========
show me top 3 customers

========== GENERATED SQL ==========
SELECT TOP (3)
    c.CUSTOMER_KEY,
    c.CUSTOMER_NAME,
    SUM(f.CURRENT_PRINCIPAL_BALANCE) AS TotalCurrentPrincipalBalance
FROM FIS_CL_DETAIL_FACT AS f
INNER JOIN FIS_MONTH_DIMENSION AS m
    ON f.MONTH_ID = m.MONTH_ID
INNER JOIN FIS_CUSTOMER_DIMENSION AS c
    ON f.CUSTOMER_KEY = c.CUSTOMER_KEY
WHERE m.MONTH_ID = (
    SELECT MAX(MONTH_ID)
    FROM FIS_MONTH_DIMENSION
)
GROUP BY
    c.CUSTOMER_KEY,
    c.CUSTOMER_NAME
ORDER BY
    TotalCurrentPrincipalBalance DESC;
