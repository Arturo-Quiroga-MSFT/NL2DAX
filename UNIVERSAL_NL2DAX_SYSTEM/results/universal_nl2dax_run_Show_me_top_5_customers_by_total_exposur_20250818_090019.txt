SELECT TOP 5
  c.CUSTOMER_ID,
  c.CUSTOMER_NAME,
  SUM(t.EXPOSURE) AS TOTAL_EXPOSURE
FROM (
  SELECT CUSTOMER_KEY, COALESCE(EXPOSURE_AT_DEFAULT, 0) AS EXPOSURE
  FROM FIS_CA_DETAIL_FACT
  UNION ALL
  SELECT CUSTOMER_KEY, COALESCE(EXPOSURE_AT_DEFAULT, 0) AS EXPOSURE
  FROM FIS_CL_DETAIL_FACT
) t
JOIN FIS_CUSTOMER_DIMENSION c
  ON t.CUSTOMER_KEY = c.CUSTOMER_KEY
GROUP BY
  c.CUSTOMER_ID,
  c.CUSTOMER_NAME
ORDER BY
  TOTAL_EXPOSURE DESC;
