========== GENERATED SQL QUERY ==========
WITH combined_exposures AS (
    SELECT TOP 5
        CUSTOMER_KEY,
        COALESCE(EXPOSURE_AT_DEFAULT, 0) AS EXPOSURE
    FROM FIS_CA_DETAIL_FACT
    UNION ALL
    SELECT
        CUSTOMER_KEY,
        COALESCE(EXPOSURE_AT_DEFAULT, 0) AS EXPOSURE
    FROM FIS_CL_DETAIL_FACT
)
SELECT
    d.CUSTOMER_ID,
    d.CUSTOMER_NAME,
    SUM(e.EXPOSURE) AS TOTAL_EXPOSURE
FROM combined_exposures e
JOIN FIS_CUSTOMER_DIMENSION d
  ON e.CUSTOMER_KEY = d.CUSTOMER_KEY
GROUP BY
    d.CUSTOMER_ID,
    d.CUSTOMER_NAME
ORDER BY
    TOTAL_EXPOSURE DESC
